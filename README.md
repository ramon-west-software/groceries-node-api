# groceries-node-api
Server side for groceries-app

dependencies:
    hot-reload for dev env:
        npm install nodemon -D
    REST API
        npm install express
    Cross origin
        npm install cors
    JWT security tokens    
        npm install jsonwebtoken
    SQL Driver for MySql database
        npm install mysql2
    Service for .env file (store keys, secrets, not for public eyes!)
        npm install dotenv
